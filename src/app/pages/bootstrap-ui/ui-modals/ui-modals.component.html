Varying Modal Content

Use event.relatedTarget and HTML data-bs-* attributes to vary the contents of the modal depending on which button was clicked.

HTML Preview
<!-- Varying Modal Content -->
<div class="hstack gap-2 flex-wrap">
<button type="button" class="btn btn-primary" (click)="varyingModal(varying,'Mary')">Open Modal for Mary</button>
<button type="button" class="btn btn-primary" (click)="varyingModal(varying,'Jennifer')">Open Modal for
    Jennifer</button>
<button type="button" class="btn btn-primary" (click)="varyingModal(varying,'Roderick')">Open Modal for
    Roderick</button>
</div>

<!-- Varying modal content -->
<ng-template #varying class="modal fade" id="varyingcontentModal">
    <div class="modal-content">
        <div class="modal-header">
            <h5 class="modal-title" id="varyingcontentModalLabel">New message {{name}}</h5>
            <button type="button" class="btn-close" (click)="modalRef?.hide()"></button>
        </div>
        <div class="modal-body">
            <!-- <form> -->
                <div class="mb-3">
                    <label for="customer-name" class="col-form-label">Customer Name:</label>
                    <input type="text" [(ngModel)]="name" class="form-control" id="customer-name">
                </div>
                <div class="mb-3">
                    <label for="message-text" class="col-form-label">Message:</label>
                    <textarea class="form-control" id="message-text" rows="4"></textarea>
                </div>
            <!-- </form> -->
        </div>
        <div class="modal-footer">
            <button type="button" class="btn btn-light" (click)="modalRef?.hide()">Back</button>
